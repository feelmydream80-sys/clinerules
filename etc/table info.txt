CREATE TABLE IF NOT EXISTS public.data_spec_params
(
    parameter_id integer NOT NULL DEFAULT nextval('tb_data_spec_params_id_seq'::regclass),
    specification_id integer NOT NULL,
    parameter_type character varying(50) COLLATE pg_catalog."default" NOT NULL,
    parameter_name_ko character varying(255) COLLATE pg_catalog."default",
    parameter_name_en character varying(255) COLLATE pg_catalog."default",
    description text COLLATE pg_catalog."default",
    is_required boolean DEFAULT false,
    CONSTRAINT tb_data_spec_params_pkey PRIMARY KEY (parameter_id),
    CONSTRAINT fk_spec FOREIGN KEY (specification_id)
        REFERENCES public.tb_data_spec (id) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE CASCADE
)

TABLESPACE pg_default;

ALTER TABLE IF EXISTS public.data_spec_params
    OWNER to etl_user;
